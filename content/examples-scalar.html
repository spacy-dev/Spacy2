We begin with the function \[f(x)=x^{3/2}+\sin(x^{1/2}).\]

#include "fung.hh"

template <class Variable>
auto generateFunction(const Variable& x)
{
  return pow<3>(sqrt(x)) + sin(sqrt(x)); 
}

Usage is as follows:

auto f = generateFunction(variable<0>(1.));

f.template update<0>(1.) //  change function argument

auto value = f();
auto firstDerivative  = f.template d1<0>(1);
auto secondDerivative = f.template d2<0,0>(1,1);
auto thirdDerivative  = f.template d3<0,0,0>(1,1,1);
